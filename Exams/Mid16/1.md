```cpp
void deleteMiddle(node*& head,int value){
    if(head==NULL){
        return;
    }else if(head->val == value){
        node* old_head = head;
        node* new_head = head->next;
        delete old_head;
        head = new_head;
        return;
    }

    node* to_delete = head->next;
    node* prev = head;

    while (to_delete != NULL && to_delete->val != value) {
        prev = to_delete;
        to_delete = to_delete->next;
    }

    prev->next = prev->next->next;
    delete prev;
}
```
